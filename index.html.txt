    DELSU: [
      {
        question: "The chemical symbol for Sodium is?",
        options: ["Na", "S", "N", "So"],
        answer: 0,
        explanation: "Sodium is represented as Na from its Latin name Natrium."
      }
    ],
    NURSING: [
      {
        question: "What is the normal human body temperature?",
        options: ["37¬∞C", "36¬∞C", "35¬∞C", "38¬∞C"],
        answer: 0,
        explanation: "The average human body temperature is 37¬∞C."
      }
    ]
  };

  let currentMode = "";
  let currentSchool = "";
  let currentQuestion = 0;
  let score = 0;

  function startPractice(school) {
    currentMode = "practice";
    startQuiz(school);
  }

  function startExam(school) {
    currentMode = "exam";
    startQuiz(school);
  }

  function startQuiz(school) {
    currentSchool = school;
    currentQuestion = 0;
    score = 0;
    document.getElementById("quiz-section").style.display = "block";
    document.getElementById("quiz-title").textContent = `${school} - ${currentMode.toUpperCase()} MODE`;
    showQuestion();
  }

  function showQuestion() {
    const q = questions[currentSchool][currentQuestion];
    const container = document.getElementById("question-container");
    container.innerHTML = `
      <div class="question-box">
        <p><strong>Q${currentQuestion + 1}:</strong> ${q.question}</p>
        ${q.options.map((opt, i) =>
          `<label class="option">
            <input type="radio" name="option" value="${i}"> ${opt}
          </label>`).join("")}
        <div class="explanation" id="explanation"></div>
      </div>
    `;
    document.getElementById("result").innerHTML = "";
  }

  function nextQuestion() {
    const q = questions[currentSchool][currentQuestion];
    const selected = document.querySelector('input[name="option"]:checked');
    if (!selected) {
      alert("Please select an option.");
      return;
    }
    const selectedIndex = parseInt(selected.value);
    if (currentMode === "practice") {
      const explanation = document.getElementById("explanation");
      if (selectedIndex === q.answer) {
        explanation.innerHTML = "‚úÖ Correct!<br>" + q.explanation;
      } else {
        explanation.innerHTML = "‚ùå Wrong. Correct answer: " + q.options[q.answer] + "<br>" + q.explanation;
      }
      return; // wait for user to click next again
    }

    if (currentMode === "exam") {
      if (selectedIndex === q.answer) score++;
    }

    currentQuestion++;
    if (currentQuestion < questions[currentSchool].length && 
        (currentMode === "practice" || currentQuestion < 50)) {
      showQuestion();
    } else {
      finishQuiz();
    }
  }

  function finishQuiz() {
    const container = document.getElementById("question-container");
    const total = currentMode === "exam" ? 50 : questions[currentSchool].length;
    let resultHTML = `<p>You answered ${score} out of ${total} correctly.</p>`;
    if (currentMode === "exam") {
      resultHTML += `<p><strong>Final Score:</strong> ${score * 2}%</p>`;
    }
    container.innerHTML = "";
    document.getElementById("result").innerHTML = resultHTML;
  }
</script>
